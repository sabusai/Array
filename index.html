Birthday Game

 function Birthday_Game(arr,D ,M) {
 let count = 0;
  for(let i=0; i <= ((arr.length)-M); i++){
  let bar_sum = 0;
  for(let j=0; j < M; j++){
  bar_sum += arr[j+i];
  }
  if(bar_sum===D){
  count += 1;
  }
  }
  return count;
}
